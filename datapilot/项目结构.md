# 项目结构
EYESURE/datapilot/
├── datapilot/                  # 项目主配置目录
│   ├── __init__.py             # 包初始化文件
│   ├── asgi.py                 # ASGI配置文件
│   ├── settings.py             # 项目设置
│   ├── urls.py                 # 项目URL配置
│   └── wsgi.py                 # WSGI配置文件
├── api/                        # API应用目录
│   ├── __init__.py             # 包初始化文件
│   ├── admin.py                # 后台管理配置
│   ├── apps.py                 # 应用配置
│   ├── models.py               # 数据模型
│   ├── serializers.py          # 数据序列化器(使用DRF)
│   ├── views.py                # 视图函数
│   ├── urls.py                 # API路由配置
│   ├── services/               # API服务层
│   │   ├── __init__.py
│   │   ├── base_client.py      # API客户端基类
│   │   ├── api_service1.py     # 特定API服务1
│   │   └── api_service2.py     # 特定API服务2
│   └── tests/                  # 测试文件目录
│       ├── __init__.py
│       └── test_apis.py        # API测试
├── utils/                      # 工具函数目录
│   ├── __init__.py
│   ├── request_utils.py        # 请求工具
│   ├── data_processing.py      # 数据处理工具
│   └── auth_utils.py           # 认证工具
├── templates/                  # 模板文件目录
├── static/                     # 静态文件目录
│   ├── css/
│   ├── js/
│   └── images/
├── media/                      # 媒体文件目录
├── .env                        # 环境变量配置文件
├── .gitignore                  # Git忽略文件
├── requirements.txt            # 项目依赖
└── manage.py                   # Django管理脚本
